-- 1. Tạo Database mới
CREATE DATABASE AirportManagement;
GO
USE AirportManagement;
GO

-- PHẦN 1: TẠO CÁC BẢNG (TABLES)

-- Bảng Vai trò (Phân quyền)
CREATE TABLE Roles (
    RoleID INT PRIMARY KEY,
    RoleName NVARCHAR(50) NOT NULL -- Admin, Staff, Passenger
);

-- Bảng Người dùng (Kien-Lam)
CREATE TABLE Users (
    UserID INT PRIMARY KEY IDENTITY(1,1),
    Username VARCHAR(50) UNIQUE NOT NULL,
    Password VARCHAR(255) NOT NULL,
    FullName NVARCHAR(100),
    RoleID INT FOREIGN KEY REFERENCES Roles(RoleID)
);

-- Bảng Chuyến bay (Tien-Lam)
CREATE TABLE Flights (
    FlightID INT PRIMARY KEY IDENTITY(1,1), --chuyen bay
    FlightCode VARCHAR(20) UNIQUE NOT NULL, --ma chuyen bay
    Departure NVARCHAR(100) NOT NULL,       --diem di
    Arrival NVARCHAR(100) NOT NULL,         --diem den
    DepartureTime DATETIME NOT NULL,        --thoi gian khoi hanh(di)
    TotalSeats INT DEFAULT 100,             --tong so ghe
    Status NVARCHAR(50) DEFAULT N'Scheduled' -- Scheduled, Boarding, Delayed, Cancelled
);

-- Bảng Danh sách ghế (Nam-Lam)
CREATE TABLE Seats (
    SeatID INT PRIMARY KEY IDENTITY(1,1),
    FlightID INT FOREIGN KEY REFERENCES Flights(FlightID),
    SeatNumber VARCHAR(10) NOT NULL, -- Ví dụ: A1, A2, B1...
    IsOccupied BIT DEFAULT 0 -- 0: Ghế trống, 1: Đã có người ngồi
);

-- Bảng Vé (Duy,Nam-Lam)
CREATE TABLE Tickets (
    TicketID INT PRIMARY KEY IDENTITY(1,1),                --ma so ve
    FlightID INT FOREIGN KEY REFERENCES Flights(FlightID), --ma chuyen bay
    PassengerID INT FOREIGN KEY REFERENCES Users(UserID),  --ma hanh khach
    SeatID INT FOREIGN KEY REFERENCES Seats(SeatID),       --Liên kết tới ghế cụ thể
    BookingDate DATETIME DEFAULT GETDATE(),                --ngay dat ve
    IsCheckedIn BIT DEFAULT 0, -- 0: Chưa check-in, 1: Đã xong thủ tục
    Status NVARCHAR(50) DEFAULT N'Booked' -- Booked, Cancelled
);
